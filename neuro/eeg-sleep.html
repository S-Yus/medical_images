R Code (ggplot2)

library(ggplot2)

# ---- EEG (Sleep Stages) ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Time (hr)", limits = c(0, 8)) +
  scale_y_continuous(name = "Sleep stage", limits = c(0, 5)) +
  labs(title = "EEG (Sleep Stages)") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
