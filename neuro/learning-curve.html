R Code (ggplot2)

library(ggplot2)

# ---- Learning/Memory Curve ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Trial number", limits = c(1, 20)) +
  scale_y_continuous(name = "Performance (% correct)", limits = c(0, 100)) +
  labs(title = "Learning/Memory Curve") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
