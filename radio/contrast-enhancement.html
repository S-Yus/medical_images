R Code (ggplot2)

library(ggplot2)

# ---- Contrast Enhancement ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Time (s)", limits = c(0, 180)) +
  scale_y_continuous(name = "Enhancement (HU)", limits = c(0, 200)) +
  labs(title = "Contrast Enhancement") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
