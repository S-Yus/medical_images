R Code (ggplot2)

library(ggplot2)

# ---- SUV Scatter Plot ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "SUVmax (Pre)", limits = c(0, 20)) +
  scale_y_continuous(name = "SUVmax (Post)", limits = c(0, 20)) +
  labs(title = "SUV Scatter Plot") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
