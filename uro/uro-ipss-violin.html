# ---- IPSS Score Violin ----

  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Treatment", limits = c(0, 3)) +
  scale_y_continuous(name = "IPSS Score", limits = c(0, 35)) +
  labs(title = "IPSS Score Violin") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
