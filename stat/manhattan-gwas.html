R Code (ggplot2)

library(ggplot2)

# ---- GWAS Manhattan Plot ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Chromosome", limits = c(0, 1)) +
  scale_y_continuous(name = "-log10(p)", limits = c(0, 15)) +
  labs(title = "GWAS Manhattan Plot") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
