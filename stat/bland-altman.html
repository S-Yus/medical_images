R Code (ggplot2)

library(ggplot2)

# ---- Bland-Altman Plot ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Mean of Two Methods", limits = c(0, 200)) +
  scale_y_continuous(name = "Difference (Method 1 - Method 2)", limits = c(-40, 40)) +
  labs(title = "Bland-Altman Plot") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
