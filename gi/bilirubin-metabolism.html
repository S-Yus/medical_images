library(ggplot2)

# ---- Bilirubin Metabolism ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Stage", limits = c(0, 5)) +
  scale_y_continuous(name = "Bilirubin (mg/dL)", limits = c(0, 20)) +
  labs(title = "Bilirubin Metabolism") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
