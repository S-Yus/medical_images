R Code (ggplot2)

library(ggplot2)

# ---- Risk Ratio with CI ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Study", limits = c(0, 5)) +
  scale_y_continuous(name = "Risk Ratio (95% CI)", limits = c(0, 5)) +
  labs(title = "Risk Ratio with CI") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
