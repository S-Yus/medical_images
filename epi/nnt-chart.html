R Code (ggplot2)

library(ggplot2)

# ---- NNT Chart ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Treatment", limits = c(0, 6)) +
  scale_y_continuous(name = "NNT", limits = c(0, 50)) +
  labs(title = "NNT Chart") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
