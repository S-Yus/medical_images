R Code (ggplot2)

library(ggplot2)

# ---- Case-Control Study ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Exposure category", limits = c(0, 4)) +
  scale_y_continuous(name = "Proportion (%)", limits = c(0, 100)) +
  labs(title = "Case-Control Study") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
