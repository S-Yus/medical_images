R Code (ggplot2)

library(ggplot2)

# ---- Allometric Growth Curve ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Body Mass (kg)", limits = c(0, 80)) +
  scale_y_continuous(name = "Organ Mass (g)", limits = c(0, 2000)) +
  labs(title = "Allometric Growth Curve") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
