R Code (ggplot2)

library(ggplot2)

# ---- Protein Denaturation Curve ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Temperature (C)", limits = c(20, 90)) +
  scale_y_continuous(name = "Fraction folded", limits = c(0, 1)) +
  labs(title = "Protein Denaturation Curve") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
