R Code (ggplot2)

library(ggplot2)

# ---- Concentration Scatter Plot ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Standard Concentration", limits = c(0, 100)) +
  scale_y_continuous(name = "Measured Absorbance", limits = c(0, 2)) +
  labs(title = "Concentration Scatter Plot") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
