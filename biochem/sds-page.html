R Code (ggplot2)

library(ggplot2)

# ---- SDS-PAGE Molecular Weight ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Rf (relative migration)", limits = c(0, 1)) +
  scale_y_continuous(name = "log MW (kDa)", limits = c(1, 2.5)) +
  labs(title = "SDS-PAGE Molecular Weight") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
