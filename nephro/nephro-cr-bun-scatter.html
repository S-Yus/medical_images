R Code (ggplot2)

library(ggplot2)

# ---- Cr vs BUN Scatter ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Serum Cr (mg/dL)", limits = c(0, 10)) +
  scale_y_continuous(name = "BUN (mg/dL)", limits = c(0, 100)) +
  labs(title = "Cr vs BUN Scatter") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
