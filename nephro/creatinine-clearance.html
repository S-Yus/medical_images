R Code (ggplot2)

library(ggplot2)

# ---- Creatinine Clearance ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Time", limits = c(0, 10)) +
  scale_y_continuous(name = "Serum creatinine (mg/dL)", limits = c(0, 10)) +
  labs(title = "Creatinine Clearance") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
