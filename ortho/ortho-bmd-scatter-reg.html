R Code (ggplot2)

library(ggplot2)

# ---- BMD vs T-score Scatter Regression ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "T-score", limits = c(-4, 2)) +
  scale_y_continuous(name = "BMD (g/cm²)", limits = c(0.4, 1.4)) +
  labs(title = "BMD vs T-score Scatter Regression") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
