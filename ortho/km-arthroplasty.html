R Code (ggplot2)

library(ggplot2)

# ---- Joint Replacement Survival ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Time (years)", limits = c(0, 20)) +
  scale_y_continuous(name = "Implant Survival", limits = c(0, 1)) +
  labs(title = "Joint Replacement Survival") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
