R Code (ggplot2)

library(ggplot2)

# ---- O2 Dissociation Curve ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "PO2 (mmHg)", limits = c(0, 100)) +
  scale_y_continuous(name = "O2 saturation (%)", limits = c(0, 100)) +
  labs(title = "O2 Dissociation Curve") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
