R Code (ggplot2)

library(ggplot2)

# ---- Nerve Action Potential ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Time (ms)", limits = c(0, 5)) +
  scale_y_continuous(name = "Membrane potential (mV)", limits = c(-90, 50)) +
  labs(title = "Nerve Action Potential") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
