R Code (ggplot2)

library(ggplot2)

# ---- Noncompetitive Antagonist ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "log [Agonist] (M)", limits = c(-9, -3)) +
  scale_y_continuous(name = "Response (% max)", limits = c(0, 100)) +
  labs(title = "Noncompetitive Antagonist") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
