R Code (ggplot2)

library(ggplot2)

# ---- Gene Expression Violin ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Sample", limits = c(0, 4)) +
  scale_y_continuous(name = "Expression (TPM)", limits = c(0, 1000)) +
  labs(title = "Gene Expression Violin") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
