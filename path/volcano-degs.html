R Code (ggplot2)

library(ggplot2)

# ---- DEG Volcano Plot ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "log2(Fold Change)", limits = c(-6, 6)) +
  scale_y_continuous(name = "-log10(p)", limits = c(0, 10)) +
  labs(title = "DEG Volcano Plot") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
