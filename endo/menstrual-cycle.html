R Code (ggplot2)

library(ggplot2)

# ---- Menstrual Cycle Hormones ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Day of cycle", limits = c(1, 28)) +
  scale_y_continuous(name = "Hormone level", limits = c(0, 10)) +
  labs(title = "Menstrual Cycle Hormones") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
