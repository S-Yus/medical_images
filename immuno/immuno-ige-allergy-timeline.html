library(ggplot2)

# ---- IgE-Mediated Allergy Reaction Timeline ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Time (hours)", limits = c(0, 24)) +
  scale_y_continuous(name = "Reaction Intensity", limits = c(0, 1)) +
  labs(title = "IgE-Mediated Allergy Reaction Timeline") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
