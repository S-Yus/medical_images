R Code (ggplot2)

library(ggplot2)

# ---- Gestational Growth Chart ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Gestational age (weeks)", limits = c(22, 42)) +
  scale_y_continuous(name = "Weight (g)", limits = c(0, 5000)) +
  labs(title = "Gestational Growth Chart") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
