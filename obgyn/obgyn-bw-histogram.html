  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Birth Weight (g)", limits = c(1500, 5000)) +
  scale_y_continuous(name = "Frequency (度数)", limits = c(0, 40)) +
  labs(title = "Birth Weight Histogram") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
