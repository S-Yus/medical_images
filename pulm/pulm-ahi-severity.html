# ---- AHI Severity Sleep Architecture Comparison ----
# あなたのデータを入力してください

  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Time (hours)", limits = c(0, 8)) +
  scale_y_continuous(name = "Sleep Stage", limits = c(0.5, 5.5)) +
  labs(title = "AHI Severity Sleep Architecture Comparison") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
