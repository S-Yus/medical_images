R Code (ggplot2)

library(ggplot2)

# ---- V/Q Diagram ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "V/Q Ratio", limits = c(0, 3)) +
  scale_y_continuous(name = "PO2 / PCO2 (mmHg)", limits = c(0, 120)) +
  labs(title = "V/Q Diagram") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
