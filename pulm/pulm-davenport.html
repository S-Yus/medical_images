library(ggplot2)

# ---- Respiratory Acid-Base Diagram ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "pH", limits = c(7, 7.7)) +
  scale_y_continuous(name = "HCO₃⁻ (mEq/L)", limits = c(5, 45)) +
  labs(title = "Respiratory Acid-Base Diagram") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
