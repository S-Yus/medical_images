R Code (ggplot2)

library(ggplot2)

# ---- Phonocardiogram ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Time (sec)", limits = c(0, 2)) +
  scale_y_continuous(name = "Amplitude", limits = c(-1, 1)) +
  labs(title = "Phonocardiogram") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
