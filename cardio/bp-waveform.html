R Code (ggplot2)

library(ggplot2)

# ---- Blood Pressure Waveform ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Time (s)", limits = c(0, 5)) +
  scale_y_continuous(name = "Pressure (mmHg)", limits = c(0, 160)) +
  labs(title = "Blood Pressure Waveform") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
