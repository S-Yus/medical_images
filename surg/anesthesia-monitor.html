R Code (ggplot2)

library(ggplot2)

# ---- Anesthesia Monitoring ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Time (min)", limits = c(0, 120)) +
  scale_y_continuous(name = "Value", limits = c(0, 200)) +
  labs(title = "Anesthesia Monitoring") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
