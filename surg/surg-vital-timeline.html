library(ggplot2)

# ---- Surgical Vital Signs Record ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Time (hours)", limits = c(0, 6)) +
  scale_y_continuous(name = "", limits = c(0, 4)) +
  labs(title = "Surgical Vital Signs Record") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
