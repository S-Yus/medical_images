R Code (ggplot2)

library(ggplot2)

# ---- Reticulocyte Scatter ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Reticulocyte (%)", limits = c(0, 20)) +
  scale_y_continuous(name = "Hb (g/dL)", limits = c(4, 18)) +
  labs(title = "Reticulocyte Scatter") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
