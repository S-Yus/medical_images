R Code (ggplot2)

library(ggplot2)

# ---- Reticulocyte Count ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Day", limits = c(0, 14)) +
  scale_y_continuous(name = "Reticulocyte count (%)", limits = c(0, 20)) +
  labs(title = "Reticulocyte Count") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
