R Code (ggplot2)

library(ggplot2)

# ---- Weight Histogram ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Weight (kg)", limits = c(0, 80)) +
  scale_y_continuous(name = "Frequency (度数)", limits = c(0, 30)) +
  labs(title = "Weight Histogram") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
