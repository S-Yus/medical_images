R Code (ggplot2)

library(ggplot2)

# ---- Weight vs Height Scatter Regression ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Height (cm)", limits = c(50, 180)) +
  scale_y_continuous(name = "Weight (kg)", limits = c(3, 80)) +
  labs(title = "Weight vs Height Scatter Regression") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
