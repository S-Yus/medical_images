library(ggplot2)

# ---- Height by Age Group (Boxplot) ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Age Group", limits = c(0, 6)) +
  scale_y_continuous(name = "Height (cm)", limits = c(50, 180)) +
  labs(title = "Height by Age Group (Boxplot)") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
