R Code (ggplot2)

library(ggplot2)

# ---- Vital Signs by Age ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Age (years)", limits = c(0, 18)) +
  scale_y_continuous(name = "Value", limits = c(0, 200)) +
  labs(title = "Vital Signs by Age") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
