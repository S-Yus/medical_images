R Code (ggplot2)

library(ggplot2)

# ---- QC Chart (Levey-Jennings) ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Day", limits = c(1, 30)) +
  scale_y_continuous(name = "QC value", limits = c(-4, 4)) +
  labs(title = "QC Chart (Levey-Jennings)") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
