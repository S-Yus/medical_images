library(ggplot2)

# ---- Method Comparison Scatter ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Method A", limits = c(0, 200)) +
  scale_y_continuous(name = "Method B", limits = c(0, 200)) +
  labs(title = "Method Comparison Scatter") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
