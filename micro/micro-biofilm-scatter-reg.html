R Code (ggplot2)

library(ggplot2)

# ---- Biofilm-OD Scatter Regression ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "OD570 (Biofilm)", limits = c(0, 3)) +
  scale_y_continuous(name = "CFU/mL", limits = c(0, 1e+08)) +
  labs(title = "Biofilm-OD Scatter Regression") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
