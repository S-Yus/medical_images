R Code (ggplot2)

library(ggplot2)

# ---- Growth Curve (Log Phase) ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Time (hr)", limits = c(0, 8)) +
  scale_y_continuous(name = "ln(OD600)", limits = c(-4, 2)) +
  labs(title = "Growth Curve (Log Phase)") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
