R Code (ggplot2)

library(ggplot2)

# ---- Antidepressant Meta-analysis ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Standardized Mean Difference", limits = c(-2, 3)) +
  scale_y_continuous(name = "", limits = c(0, 8)) +
  labs(title = "Antidepressant Meta-analysis") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
