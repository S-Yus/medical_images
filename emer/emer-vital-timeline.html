R Code (ggplot2)

library(ggplot2)

# ---- Vital Signs Timeline ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Time (hours)", limits = c(0, 24)) +
  scale_y_continuous(name = "", limits = c(0, 4)) +
  labs(title = "Vital Signs Timeline") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
