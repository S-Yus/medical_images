R Code (ggplot2)

library(ggplot2)

# ---- Shock Classification ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Class", limits = c(0, 5)) +
  scale_y_continuous(name = "Blood loss (mL) / HR / BP", limits = c(0, 2000)) +
  labs(title = "Shock Classification") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
