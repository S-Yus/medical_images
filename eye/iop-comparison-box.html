R Code (ggplot2)

library(ggplot2)

# ---- IOP by Glaucoma Type (Boxplot) ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Type", limits = c(0, 4)) +
  scale_y_continuous(name = "IOP (mmHg)", limits = c(0, 60)) +
  labs(title = "IOP by Glaucoma Type (Boxplot)") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
