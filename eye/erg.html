R Code (ggplot2)

library(ggplot2)

# ---- Electroretinogram ----
# あなたのデータを入力してください
df <- data.frame(
  x = c(),
  y = c()
)

ggplot(df, aes(x = x, y = y)) +
  geom_point(size = 2) +
  geom_line() +
  scale_x_continuous(name = "Time (ms)", limits = c(0, 200)) +
  scale_y_continuous(name = "Amplitude (uV)", limits = c(-100, 500)) +
  labs(title = "Electroretinogram") +
  theme_bw(base_size = 14) +
  theme(plot.title = element_text(face = "bold", hjust = 0.5))
